import React, { useState } from 'react';
import { useNavigate, useLocation } from 'react-router-dom';
import { useAuthStore } from '../stores/auth';

// 默认邮箱域名后缀
const EMAIL_DOMAIN = 'openfaith.app';

function Login() {
  const navigate = useNavigate();
  const location = useLocation();
  const authStore = useAuthStore();
  const [loginType, setLoginType] = useState<'email' | 'account'>('email');
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  
  const [showForgotModal, setShowForgotModal] = useState(false);
  const [forgotEmail, setForgotEmail] = useState('');
  const [resetToken, setResetToken] = useState('');
  const [newPassword, setNewPassword] = useState('');
  const [forgotStep, setForgotStep] = useState(1);
  const [isSendingCode, setIsSendingCode] = useState(false);

  // 将输入转换为邮箱格式
  const normalizeEmail = (input: string): string => {
    const trimmed = input.trim();
    // 如果已经是邮箱格式，直接返回
    if (trimmed.includes('@')) {
      return trimmed;
    }
    // 否则拼接域名后缀
    return `${trimmed}@${EMAIL_DOMAIN}`;
  };

  const handleLogin = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!email || !password) {
      alert('请填写账号/邮箱和密码');
      return;
    }

    setIsLoading(true);
    try {
      // 账号登录模式：将输入转为邮箱格式
      const loginEmail = loginType === 'account' ? normalizeEmail(email) : email;
      const result = await authStore.login(loginEmail, password);
      setIsLoading(false);

      if (result.success) {
        console.log('[Login] 登录成功，跳转首页');
        // 登录成功后直接跳转到首页
        window.location.hash = '#/';
      } else {
        alert('登录失败：' + result.error);
      }
    } catch (err) {
      setIsLoading(false);
      alert('登录异常：' + (err as Error).message);
    }
  };

  const handleSendResetLink = async () => {
    if (!forgotEmail) {
      alert('请输入邮箱');
      return;
    }
    
    setIsSendingCode(true);
    try {
      const response = await fetch('/sb-api/auth/v1/forgot_password', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email: forgotEmail })
      });
      const data = await response.json();
      
      setIsSendingCode(false);
      
      if (response.ok) {
        // 演示模式：直接显示重置token
        if (data.demo_token) {
          alert(`重置链接已生成（演示模式）\n\n邮箱: ${forgotEmail}\n重置令牌: ${data.demo_token}\n\n请复制上面的令牌，在下方输入完成密码重置。`);
        } else {
          alert(data.message || '重置链接已发送到邮箱');
        }
        setForgotStep(2);
      } else {
        alert('发送失败：' + (data.msg || data.message || '未知错误'));
      }
    } catch (err) {
      setIsSendingCode(false);
      alert('发送失败：' + (err as Error).message);
    }
  };

  const handleResetPassword = async () => {
    if (!resetToken || !newPassword) {
      alert('请填写重置令牌和新密码');
      return;
    }
    
    if (newPassword.length < 6) {
      alert('新密码至少需要6位');
      return;
    }
    
    setIsLoading(true);
    try {
      const response = await fetch('/sb-api/auth/v1/reset_password', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ 
          email: forgotEmail, 
          password: newPassword,
          token: resetToken 
        })
      });
      
      setIsLoading(false);
      
      if (response.ok) {
        alert('密码重置成功！请使用新密码登录');
        setShowForgotModal(false);
        setForgotStep(1);
        setForgotEmail('');
        setResetToken('');
        setNewPassword('');
      } else {
        const data = await response.json();
        alert('重置失败：' + (data.msg || data.message || '未知错误'));
      }
    } catch (err) {
      setIsLoading(false);
      alert('重置失败：' + (err as Error).message);
    }
  };

  const handleClearUsers = async () => {
    if (!confirm('确定要清除所有用户数据吗？这将删除所有注册账号。')) {
      return;
    }
    
    try {
      const response = await fetch('/sb-api/auth/v1/admin/clear_users', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' }
      });
      
      if (response.ok) {
        alert('所有用户数据已清除！现在可以重新注册。');
        setShowForgotModal(false);
      } else {
        alert('清除失败');
      }
    } catch (err) {
      alert('清除失败：' + (err as Error).message);
    }
  };

  return (
    

      

        
登录

        
        {/* 登录方式切换 */}
        

          
          
        

        
        

          

            
             setEmail(e.target.value)}
              placeholder={loginType === 'email' ? '请输入邮箱...' : '请输入账号...'}
              className="w-full h-12 px-4 rounded-xl border border-gray-200 text-sm focus:border-[#E11D48] focus:outline-none"
            />
            {loginType === 'account' && (
              

                自动转换为 {EMAIL_DOMAIN} 邮箱
              

            )}
          

          
          

            
             setPassword(e.target.value)}
              placeholder="请输入密码..."
              className="w-full h-12 px-4 rounded-xl border border-gray-200 text-sm focus:border-[#E11D48] focus:outline-none"
            />
          

          
          
        

        
        

          
        

        
        

          
还没有账号？

          
        

      


      {showForgotModal && (
        
 setShowForgotModal(false)}>
          
 e.stopPropagation()}>
            

              {forgotStep === 1 ? '找回密码' : '重置密码'}
            

            
            
            
            {forgotStep === 1 ? (
              

                 setForgotEmail(e.target.value)}
                  placeholder="请输入注册邮箱"
                  className="w-full h-12 px-4 rounded-xl border border-gray-200 text-sm focus:border-[#E11D48] focus:outline-none"
                />
                
              

            ) : (
              

                 setResetToken(e.target.value)}
                  placeholder="请输入重置令牌"
                  className="w-full h-12 px-4 rounded-xl border border-gray-200 text-sm focus:border-[#E11D48] focus:outline-none"
                />
                 setNewPassword(e.target.value)}
                  placeholder="请输入新密码（至少6位）"
                  className="w-full h-12 px-4 rounded-xl border border-gray-200 text-sm focus:border-[#E11D48] focus:outline-none"
                />
                
              

            )}
            
            
          

        

      )}
    

  );
}

export default Login;
